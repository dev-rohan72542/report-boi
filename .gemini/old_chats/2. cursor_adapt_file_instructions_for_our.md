```typescriptreact
"use client"

import { Card, CardContent } from "@/components/ui/card"
import { TrendingUp, TrendingDown } from "lucide-react"

interface MetricCardProps {
  label: string
  value: React.ReactNode
  sparkline?: React.ReactNode
  trend?: 'up' | 'down' | 'neutral'
  trendValue?: string
  unit?: string
}

export function MetricCard({ 
  label, 
  value, 
  sparkline, 
  trend, 
  trendValue, 
  unit 
}: MetricCardProps) {
  return (
    <div className="card-glass p-4 md:p-5 flex flex-col gap-3 group hover:shadow-glow transition-all duration-300">
      <div className="text-xs uppercase tracking-wide text-muted font-medium">{label}</div>
      <div className="flex items-end gap-2">
        <div className="text-3xl font-semibold text-foreground">{value}</div>
        {unit && <span className="text-sm text-muted mb-1">{unit}</span>}
      </div>
      {trend && trendValue && (
        <div className="flex items-center gap-1 text-xs">
          {trend === 'up' ? (
            <TrendingUp className="h-3 w-3 text-green-400" />
          ) : trend === 'down' ? (
            <TrendingDown className="h-3 w-3 text-red-400" />
          ) : null}
          <span className={trend === 'up' ? 'text-green-400' : trend === 'down' ? 'text-red-400' : 'text-muted'}>
            {trendValue}
          </span>
        </div>
      )}
      {sparkline && (
        <div className="mt-2 opacity-60 group-hover:opacity-100 transition-opacity">
          {sparkline}
        </div>
      )}
    </div>
  )
}
```

```typescriptreact
import { redirect } from "next/navigation"
import { createClient } from "@/lib/supabase/server"
import { DailyEntryForm } from "@/components/daily-entry-form"
import { MetricCard } from "@/components/metric-card"
import { BookOpen, Briefcase, Dumbbell, Users } from "lucide-react"

export default async function DashboardPage() {
  const supabase = await createClient()
  const { data, error } = await supabase.auth.getUser()

  if (error || !data?.user) {
    redirect("/auth/login")
  }

  // Get today's entry if it exists
  const today = new Date().toISOString().split("T")[0]
  const { data: todayEntry } = await supabase
    .from("daily_entries")
    .select("*")
    .eq("user_id", data.user.id)
    .eq("entry_date", today)
    .single()

  // Get last 7 days of entries for metrics
  const sevenDaysAgo = new Date()
  sevenDaysAgo.setDate(sevenDaysAgo.getDate() - 7)

  const { data: recentEntries } = await supabase
    .from("daily_entries")
    .select("*")
    .eq("user_id", data.user.id)
    .gte("entry_date", sevenDaysAgo.toISOString().split("T")[0])
    .order("entry_date", { ascending: true })

  // Calculate KPIs
  const kpis = {
    workTotal: recentEntries?.reduce((sum, entry) => sum + (entry.online_work || 0) + (entry.offline_work || 0), 0) || 0,
    quranTotal: recentEntries?.reduce((sum, entry) => sum + (entry.quran_study || 0), 0) || 0,
    exerciseTotal: recentEntries?.reduce((sum, entry) => sum + (entry.pull_ups || 0) + (entry.push_ups || 0) + (entry.squats || 0), 0) || 0,
    communicationTotal: recentEntries?.reduce((sum, entry) => 
      sum + (entry.communication_members || 0) + (entry.communication_companions || 0) + 
      (entry.communication_workers || 0) + (entry.communication_relations || 0) + 
      (entry.communication_friends || 0) + (entry.communication_talented_students || 0) + 
      (entry.communication_well_wishers || 0) + (entry.communication_mahram || 0), 0) || 0,
  }

  return (
    <div className="space-y-6">
      {/* Metrics Grid */}
      <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-3 md:gap-4 mb-6">
        <MetricCard 
          label="Work Hours" 
          value={Math.round(kpis.workTotal / 60 * 10) / 10} 
          unit="hrs"
        />
        <MetricCard 
          label="Qur'an Study" 
          value={kpis.quranTotal} 
          unit="min"
        />
        <MetricCard 
          label="Exercise" 
          value={kpis.exerciseTotal} 
          unit="reps"
        />
        <MetricCard 
          label="Contacts" 
          value={kpis.communicationTotal} 
          unit="people"
        />
      </div>

      {/* Daily Entry Form */}
      <div className="card-glass p-6">
        <div className="mb-6">
          <h1 className="text-2xl md:text-3xl font-bold text-foreground mb-2">আজকের এন্ট্রি</h1>
          <p className="text-sm md:text-base text-muted">
            {new Date().toLocaleDateString("bn-BD", {
              weekday: "long",
              year: "numeric",
              month: "long",
              day: "numeric",
            })}
          </p>
        </div>
        <DailyEntryForm initialData={todayEntry} userId={data.user.id} />
      </div>
    </div>
  )
}
```
